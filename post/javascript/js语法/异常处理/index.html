throw语句

function fn(x,y) {

    if (!x || !y) {
        throw new Error("参数错误")
    }
    return x + y
}
console.log(fn())

try…catch 语句

try {

    console.log(111)
  	//调用不存在的方法
    abc()
    console.log(222)

} catch (e) {
    console.log("异常:",e.message)
}

console.log(333)
//打印结果
//111
//异常: abc is not defined
//333

断点添加

function fn(x,y) {

    console.log(111)
    //添加断点
    debugger
    console.log(111)
    console.log(111)

}
console.log(fn())
